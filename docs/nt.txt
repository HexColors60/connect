basic usage:

-- compile:
> build

-- run:
- prepare service:

> sc create PacketFilter type= kernel binpath= C:\Users\user\Development\kernel\filter\objchk_win7_amd64\amd64\sys_nt.sys

- run:

> sc start ndsniff

- stop:

> sc stop ndsniff

- remove:

> sc delete ndsniff
